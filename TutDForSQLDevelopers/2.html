SELECT 
t.table_name,
array_agg((column_name::text,c.data_type::text)) AS column_info FROM information_schema.tables AS t 
JOIN information_schema.columns as c on c.table_name=t.table_name 
AND c.table_schema=t.table_schema 
WHERE t.table_schema='public' group by t.table_name;
